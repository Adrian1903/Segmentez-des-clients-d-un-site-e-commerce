# Initialisation
columns = ['order_id',
           'order_purchase_timestamp',
           'delivery_respect',
           'total_paid',
           'review_score']

columns_rename = {'order_purchase_timestamp': 'recency',
                  'order_id': 'frequency',
                  'total_paid': 'monetary_value'}

aggdict = {'order_id': 'count',
           'order_purchase_timestamp': lambda x: (snapshot_date
                                                  - x.max()).days,
           'delivery_respect': lambda x: x.value_counts().index[0],
           'total_paid': 'sum',
           'review_score': lambda x: x.value_counts().index[0]}

columns_flow = ['customer_unique_id', 'period', 'cluster_name']
columns_index = ['review_score', 'monetary_value_log']
list = ['recency', 'frequency', 'monetary_value']

sales_flow = pd.DataFrame()
customer_flow = pd.DataFrame()

scaler = StandardScaler()
kmeans = KMeans(n_clusters=6, random_state=123)

for p in list_period:
    # Ajout du trimestre
    sales_flow = sales[sales.period <= p]
    print(f'Traitement de {p} et ant√©rieur')
    # Calcul des m√©triques
    customer_temp = sales_flow.groupby('customer_unique_id')[columns]\
                              .agg(aggdict)
    customer_temp.rename(columns=columns_rename, inplace=True)
    print(f'Shape period : {customer_temp.shape}')
    # Passage au log des m√©triques RFM
    for i in list:
        customer_temp[i + '_log'] = np.log(customer_temp[i])
        customer_temp.drop(i, axis=1, inplace=True)
    # Standardisation des donn√©es
    std_customer_temp = scaler.fit_transform(customer_temp)
    std_customer_temp = pd.DataFrame(std_customer_temp,
                                     index=customer_temp.index,
                                     columns=customer_temp.columns)
    # Kmeans
    kmeans.fit(std_customer_temp)
    # Int√©gration des centro√Ødes dans un dataframe
    ctds = pd.DataFrame(kmeans.cluster_centers_,
                        columns=std_customer_temp.columns,
                        index=(range(0, 6)))
    ctds = ctds.reset_index().rename(columns={'index': 'cluster'})

    # R√©cup√©ration des sp√©cificit√©s des centro√Ødes
    # pour le nommage des clusters
    several_orders = ctds.sort_values(by='frequency_log')\
                         .tail(1).cluster.iloc[0]

    recent_order = ctds.sort_values(by='monetary_value_log')\
                       .head(2).cluster.iloc[1]

    low_value = ctds.sort_values(by='monetary_value_log')\
                    .head(1).cluster.iloc[0]

    c_fix = ['delivery_respect', 'review_score']
    ctds['fix_customer'] = ctds[c_fix].sum(axis=1)
    bad_score = ctds.sort_values(by='fix_customer')\
                    .head(2).cluster.iloc[1]

    ctds['fix_customer'] = ctds[c_fix].sum(axis=1)
    bad_delivery = ctds.sort_values(by='fix_customer')\
                       .head(1).cluster.iloc[0]

    c_top = ['review_score', 'monetary_value_log']
    ctds['top_customer'] = ctds[c_top].sum(axis=1)
    top_customer = ctds.sort_values(by='top_customer')\
                       .tail(1).cluster.iloc[0]

    # Mise √† jour du dictionnaire des clusters
    dict_cluster = {'2.Client actif': several_orders,
                    '5.Client non livr√©/en retard': bad_delivery,
                    '6.Client m√©content': bad_score,
                    '3.Client r√©cent': recent_order,
                    '4.Client √† faible valeur': low_value,
                    '1.Top client': top_customer}

    # Assignation des clusters et p√©riodes
    # sur le df principal (std_customer_temp)
    # et le df de v√©rification (ctds)
    cluster_labels = kmeans.labels_
    std_customer_temp = affect_cluster_name(std_customer_temp,
                                            cluster_labels,
                                            dict_cluster)

    std_customer_temp['period'] = p
    std_customer_temp = std_customer_temp.reset_index()

    # V√©rification dans le prompte du bon nommage des clusters
    # sert de validation de la proc√©dure
    ctds['cluster_name'] = np.nan
    for i, n in dict_cluster.items():
        ctds['cluster_name'][ctds.cluster == n] = i
    display(round(ctds, 2))

    # Concat√©nation
    customer_flow = customer_flow.append(std_customer_temp[columns_flow])
